169  float theta_in = acos(normal.to_vec().dot(inc_dir));
170  float theta_out = acos(normal.to_vec().dot(out_dir));
171 
172  if (fabs(theta_in - theta_out) < threshold_angle_rad) {
173  return (*pigment)(uv);
