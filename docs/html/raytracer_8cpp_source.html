78 void split(const string &str, const string &delim, vector<string> &parts) {
79  size_t start, end = 0;
80  while (end < str.size()) {
81  start = end;
82  while (start < str.size() && (delim.find(str[start]) != string::npos)) {
83  start++; // skip initial whitespace
84  }
85  end = start;
86  while (end < str.size() && (delim.find(str[end]) == string::npos)) {
87  end++; // skip to end of word
88  }
89  if (end - start != 0) { // just ignore zero-length strings.
90  parts.push_back(string(str, start, end - start));
91  }
92  }
93 }

102 map<string, float> build_vars_table(vector<string> &definitions) {
103  map<string, float> vars;
104  for (const auto &el : definitions) {
105  vector<string> parts;
106  split(el, ":", parts);
107  if (parts.size() != 2) {
108  fmt::print(
109  "ERROR: the definition {} does not follow the pattern NAME:VALUE\n",
110  el);
111  exit(1);
112  }
113  string name = parts.at(0), value = parts.at(1);
114  float f_value = 0.;
115  try {
116  f_value = stof(value);
117  } catch (invalid_argument &e) {
118  fmt::print("invalid floating point value {} in definition {}\n", value,
119  el);
120  }
121  vars[name] = f_value;
122  }
123 
124  return vars;
125 }

127 void imagerender(int width, int height, string algorithm, int init_state,
128  int init_seq, int num_of_rays, int max_depth,
129  int samples_per_pixel, string output_file, string input_scene,
130  vector<string> &cli_vars) {
131 
132  // Checking if antialiasing feature is on, and properly set
133  int samples_per_side = static_cast<int>(sqrt(samples_per_pixel));
134  if (pow(samples_per_side, 2) != samples_per_pixel) {
135  fmt::print("ERROR: the number of samples per pixel ({}) must be a perfect "
136  "square.\nExiting.\n",
137  samples_per_pixel);
138  exit(1);
139  }
140 
141  // Checking if user defined an output file
142  if (output_file == "")
143  throw invalid_argument("You must specify the output filename");
144 
145  string format = static_cast<string>(output_file)
146  .erase(0, static_cast<string>(output_file).find(".") + 1);
147  string name = static_cast<string>(output_file)
148  .substr(0, static_cast<string>(output_file).find("."));
149  if (format != output_file) // if user have specified a format
150  output_file = name;
151 
152  string pfm_output = output_file + ".pfm";
153  string png_output;
154  if (format == "jpeg" || format == "jpg" || format == "JPEG")
155  png_output = output_file + ".jpg";
156  else
157  png_output = output_file + ".png";
158 
159  // Parsing the input file defining the scene
160  ifstream scene_file(input_scene);
161  if (scene_file.fail()) {
162  fmt::print("ERROR: unable to open {} file\n", input_scene);
163  exit(1);
164  }
165  InputStream stream(scene_file, input_scene);
166  map<string, float> vars = build_vars_table(cli_vars);
167  Scene scene;
168  try {
169  scene = stream.parse_scene(vars);
170  } catch (GrammarError &e) {
171  fmt::print(e.what());
172  exit(1);
173  }
174 
175  /* Warn the user if the aspect_ratio specified by CLI or in input file is
176  * different from width/height */
177  float _expected_aspect_ratio = static_cast<float>(width) / height;
178  if (!are_close(scene.camera->aspect_ratio, _expected_aspect_ratio, 1e-3)) {
179  fmt::print(
180  "The aspect ratio you defined ({}) is not the ideal one ({}) for this "
181  "dimensions\nNote that the aspect ratio should be width:height\n",
182  scene.camera->aspect_ratio, _expected_aspect_ratio);
183  string answer;
184  fmt::print("Do you want to change it? [Y|N] ");
185  cin >> answer;
186  if (answer == "Y" || answer == "yes" || answer == "Yes" || answer == "y") {
187  float new_aspect_ratio;
188  fmt::print("Insert new value for aspect ratio: ");
189  cin >> new_aspect_ratio;
190  scene.camera->aspect_ratio = new_aspect_ratio;
191  } else if (answer == "N" || answer == "no" || answer == "No" ||
192  answer == "n") {
193  fmt::print("Okay, your image your choice, don't blame me if it'll come "
194  "out strange\n");
195  } else {
196  fmt::print("Answer not recognized. Exiting\n");
197  exit(1);
198  }
199  }
200  // Allocating the image
201  HdrImage image(width, height);
202 
203  // Allocating the tracer
204  ImageTracer tracer(image, scene.camera, samples_per_side);
205 
206  // Allocating the user-chosen renderer
207  shared_ptr<Renderer> renderer;
208  if (algorithm == "onoff") {
209  fmt::print("Using on/off renderer\n");
210  renderer = make_shared<OnOffRenderer>(scene.world);
211  } else if (algorithm == "flat") {
212  fmt::print("Using flat renderer\n");
213  renderer = make_shared<FlatRenderer>(scene.world);
214  } else if (algorithm == "pathtracing") {
215  fmt::print("Using a path tracer\n");
216  renderer = make_shared<PathTracer>(
217  scene.world, BLACK, PCG(init_state, init_seq), num_of_rays, max_depth);
218  } else {
219  fmt::print("Unknown renderer type.\nExiting.\n");
220  exit(1);
221  }
222 
223  Timer t;
224  // Rendering the image (time-consuming process, where the "magic" happens)
225  tracer.fire_all_rays([&](const Ray &ray) { return (*renderer)(ray); });
226  fmt::print("Rendering completed in {} s\n", t.elapsed());
227 
228  // Writing pfm file
229  ofstream pfm_stream(pfm_output);
230  tracer.image.write_pfm(pfm_stream, Endianness::little_endian);
231  fmt::print("File {} has been written to disk\n", pfm_output);
232 
233  // Apply tone - mapping to the image
234  tracer.image.normalize_image(1.0);
235  tracer.image.clamp_image();
236 
237  // Writing image in ldr format (for now png)
238  tracer.image.write_ldr_image(png_output.c_str(), 1.0);
239  fmt::print("File {} has been written to disk. \n", png_output);
240 }

253 pfm2png::pfm2png(string input_pfm_filename, string output_filename,
254  float _factor = 1., float _gamma = 1., float _luminosity = 0.)
255  : image(input_pfm_filename) {
256 
257  factor = _factor;
258  gamma = _gamma;
259  luminosity = _luminosity;
260  fmt::print("File {} has been read from disk. \n", input_pfm_filename);
261 
262  // Run Tone-Mapping
263  if (luminosity == 0.) {
264  image.normalize_image(factor);
265  } else {
266  image.normalize_image(factor, luminosity);
267  }
268  image.clamp_image();
269 
270  // Open output file
271  image.write_ldr_image(output_filename.c_str(), gamma);
272  fmt::print("File {} has been written to disk. \n", output_filename);
273 }
