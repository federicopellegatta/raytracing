22 HitRecord Sphere::ray_intersection(Ray ray) {
23  Ray inv_ray = ray.transform(transformation.inverse());
24  Vec origin_vec = inv_ray.origin.to_vec();
25  bool hit = true;
26 
27  float a = inv_ray.dir.squared_norm();
28  float b = 2.0 * origin_vec.dot(inv_ray.dir);
29  float c = origin_vec.squared_norm() - 1.0;
30 
31  float delta = b * b - 4.0 * a * c;
32  if (delta <= 0.0)
33  hit = false;
34 
35  float tmin = (-b - sqrt(delta)) / (2.0 * a);
36  float tmax = (-b + sqrt(delta)) / (2.0 * a);
37  float first_hit_t = 0.f;
38 
39  if (tmin > inv_ray.tmin && tmin < inv_ray.tmax)
40  first_hit_t = tmin;
41  else if (tmax > inv_ray.tmin && tmax < inv_ray.tmax)
42  first_hit_t = tmax;
43  else
44  hit = false;
45 
46  Point hit_point = inv_ray.at(first_hit_t);
47 
48  return HitRecord(make_shared<Sphere>(*this), transformation * hit_point,
49  transformation * sphere_normal(hit_point, ray.dir),
50  sphere_point_to_uv(hit_point), first_hit_t, ray, hit);
51 }

53 HitRecord Plane::ray_intersection(Ray ray) {
54  Ray inv_ray = ray.transform(transformation.inverse());
55  Vec origin_vec = inv_ray.origin.to_vec();
56  bool hit = true;
57 
58  if (are_close(inv_ray.dir.z, 0.0)) // if ray is parallel to the plane
59  hit = false;
60 
61  float t = -origin_vec.z / inv_ray.dir.z;
62 
63  if (t <= inv_ray.tmin || t >= inv_ray.tmax)
64  hit = false;
65 
66  Point hit_point = inv_ray.at(t);
67  return HitRecord(make_shared<Plane>(*this), transformation * hit_point,
68  transformation * plane_normal(hit_point, ray.dir),
69  plane_point_to_uv(hit_point), t, ray, hit);
70 }
