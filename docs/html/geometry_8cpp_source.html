39 ONB::ONB(Normal normal) {
40  normal.normalize();
41  float sign = copysign(1.0, normal.z);
42  float a = -1.0 / (sign + normal.z);
43  float b = normal.x * normal.y * a;
44 
45  e1 = Vec(1.0 + sign * normal.x * normal.x * a, sign * b, -sign * normal.x);
46  e2 = Vec(b, sign + normal.y * normal.y * a, -normal.y);
47  e3 = normal.to_vec();
48 }

72  for (int i{}; i < 4; i++) {

81 // Product between two 4x4 matrices
82 void _matr_prod(const float a[4][4], const float b[4][4], float c[4][4]) {
83  for (int i{}; i < 4; i++) {
84  for (int j{}; j < 4; j++) {
85  for (int k{}; k < 4; k++)

116 Point Transformation::operator*(Point p) {
117  float x = m[0][0] * p.x + m[0][1] * p.y + m[0][2] * p.z + m[0][3];
118  float y = m[1][0] * p.x + m[1][1] * p.y + m[1][2] * p.z + m[1][3];
119  float z = m[2][0] * p.x + m[2][1] * p.y + m[2][2] * p.z + m[2][3];
120  float w = m[3][0] * p.x + m[3][1] * p.y + m[3][2] * p.z + m[3][3];
121 
122  if (w != 1) {
123  x = x / w;
124  y = y / w;
125  z = z / w;
126  }
127 
128  return Point{x, y, z};
129 }
