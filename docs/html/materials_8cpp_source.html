42 Ray DiffusiveBRDF::scatter_ray(PCG &pcg, Vec inc_dir, Point interaction_point,
43  Normal normal, int depth) {
44  ONB orthonormal_base(normal);
45  float cos_theta_sq = pcg.random_float();
46  float cos_theta = sqrt(cos_theta_sq), sin_theta = sqrt(1 - cos_theta_sq);
47  float phi = 2. * M_PI * pcg.random_float();
48  Vec dir(orthonormal_base.e1 * cos(phi) * cos_theta +
49  orthonormal_base.e2 * sin(phi) * cos_theta +
50  orthonormal_base.e3 * sin_theta);
51  return Ray(interaction_point, dir, depth, 1e-3);
52 }
