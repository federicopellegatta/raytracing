22 void HdrImage::read_pfm(istream &stream) {
23  if (!stream)
24  throw ios_base::failure("File does not exist");
25 
26  // get length of file:
27  stream.seekg(0, stream.end);
28  int file_len = stream.tellg();
29  stream.seekg(0, stream.beg);
30  if (file_len == -1)
31  throw InvalidPfmFileFormat("The file is empty");
32 
33  // check file format: is it a PFM file?
34  string magic;
35  getline(stream, magic);
36  if (magic != "PF")
37  throw InvalidPfmFileFormat("Invalid magic in PFM file");
38 
39  // read img_size
40  string img_size;
41  getline(stream, img_size);
42  width = parse_img_size(img_size)[0];
43  height = parse_img_size(img_size)[1];
44 
45  // read endianness
46  string e;
47  Endianness endianness;
48  getline(stream, e);
49  endianness = parse_endianness(e);
50 
51  // check if img pixels are >= width*height
52  int header_len = stream.tellg();
53  if ((file_len - header_len) < (width * height * 3 * 4))
54  throw InvalidPfmFileFormat("Invalid file dimension");
55 
56  allocate_memory(width, height);
57  // Read the image
58  for (int y{height - 1}; y >= 0; y--) {
59  for (int x{}; x < width; x++) {
60  float r = read_float(stream, endianness);
61  float g = read_float(stream, endianness);
62  float b = read_float(stream, endianness);
63  set_pixel(x, y, Color(r, g, b));
64  }
65  }
66 }

93 void HdrImage::write_float(ostream &stream, float value,
94  Endianness endianness) {
95  // Convert "value" in a sequence of 32 bit
96  uint32_t double_word{*((uint32_t *)&value)};
97 
98  // Extract the four bytes in "double_word" using bit-level operators
99  uint8_t bytes[] = {
100  static_cast<uint8_t>(double_word & 0xFF), // Least significant byte
101  static_cast<uint8_t>((double_word >> 8) & 0xFF),
102  static_cast<uint8_t>((double_word >> 16) & 0xFF),
103  static_cast<uint8_t>((double_word >> 24) & 0xFF), // Most significant byte
104  };
105 
106  switch (endianness) {
107  case Endianness::little_endian:
108  for (int i{}; i < 4; ++i) // Forward loop
109  stream << bytes[i];
110  break;
111 
112  case Endianness::big_endian:
113  for (int i{3}; i >= 0; --i) // Backward loop
114  stream << bytes[i];
115  break;
116  }
117 }

119 void HdrImage::write_pfm(ostream &stream, Endianness e) {
120  stringstream sstr;
121 
122  float float_endianness;
123  if (e == Endianness::little_endian) {
124  float_endianness = -1.f;
125  } else {
126  float_endianness = 1.f;
127  }
128 
129  // fixed and setprecision in order to have .0 in output
130  sstr << "PF\n"
131  << width << " " << height << "\n"
132  << fixed << setprecision(1) << float_endianness;
133 
134  string result{sstr.str()};
135 
136  stream << result << endl;
137  for (int y = height - 1; y >= 0; y--) {
138  for (int x = 0; x < width; x++) {
139  Color color = get_pixel(x, y);
140  write_float(stream, color.r, e);
141  write_float(stream, color.g, e);
142  write_float(stream, color.b, e);
143  }
144  }
145 }

147 float HdrImage::read_float(istream &stream, Endianness e) {
148  unsigned char bytes[4];
149 
150  for (int i{}; i < 4; i++)
151  stream >> noskipws >> bytes[i];
152 
153  float value = 0.f;
154  if (e == Endianness::little_endian)
155  memcpy(&value, &bytes, sizeof(value));
156 
157  if (e == Endianness::big_endian) {
158  reverse(begin(bytes), end(bytes));
159  memcpy(&value, &bytes, sizeof(value));
160  }
161  return value;
162 }

165 Endianness parse_endianness(string str) {
166 
167  float floatEndianness = 0.f;
168  try {
169  floatEndianness = stof(str);
170  } catch (invalid_argument) {
171  throw InvalidPfmFileFormat("Missing endianness specification");
172  }
173 
174  if (floatEndianness == -1.0) {
175  return Endianness::little_endian;
176  } else if (floatEndianness == 1.0) {
177  return Endianness::big_endian;
178  } else {
179  throw InvalidPfmFileFormat("Invalid endianness specification");
180  }
181 }

183 vector<int> parse_img_size(string str) {
184  string w = str.substr(0, str.find(" "));
185  string h = str.erase(0, w.size() + 1).substr(0, str.find(" "));
186  string rest = str.erase(0, w.size() + 1);
187 
188  int width = 0;
189  int height = 0;
190  try {
191  if (rest == "") {
192  width = stoi(w); // nota: stoi(1.1)=1
193  height = stoi(h);
194  } else {
195  throw InvalidPfmFileFormat(
196  "Invalid image dimension: more than 2 dimensions");
197  }
198 
199  if (width < 0 || height < 0) {
200  throw InvalidPfmFileFormat(
201  "Invalid image dimension: width and/or height is negative integer");
202  }
203  } catch (invalid_argument) {
204  throw InvalidPfmFileFormat(
205  "Invalid image dimension: width and/or height not a integer");
206  }
207 
208  vector<int> dim = {width, height};
209  return dim;
210 }

238 void HdrImage::write_ldr_image(const char *output_filename, float gamma) {
239  gdImagePtr img;
240  FILE *output_file;
241 
242  img = gdImageCreateTrueColor(width, height);
243  if (gamma <= 0)
244  throw invalid_argument("GAMMA must be a positive floating point");
245 
246  for (int row{}; row < height; ++row) {
247  for (int col{}; col < width; ++col) {
248  Color cur_color = get_pixel(col, row);
249  int red, green, blue;
250  red = static_cast<int>(255 * pow(cur_color.r, 1 / gamma));
251  green = static_cast<int>(255 * pow(cur_color.g, 1 / gamma));
252  blue = static_cast<int>(255 * pow(cur_color.b, 1 / gamma));
253  gdImageSetPixel(img, col, row, gdImageColorExact(img, red, green, blue));
254  }
255  }
256 
257  output_file = fopen(output_filename, "wb");
258  if (!output_file) {
259  throw ios_base::failure("Failed to open output file.");
260  }
261 
262  string format =
263  static_cast<string>(output_filename)
264  .erase(0, static_cast<string>(output_filename).find(".") + 1);
265 
266  if (format == "png") {
267  gdImagePng(img, output_file);
268  } else if (format == "jpeg" || format == "jpg") {
269  gdImageJpeg(img, output_file, -1);
270  } else {
271  throw runtime_error("Format is not supported");
272  }
273 
274  fclose(output_file);
275  gdImageDestroy(img);
276 }
